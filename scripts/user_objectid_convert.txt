db.emaildatas.find().snapshot().forEach(
    function (doc) {
        db.emaildatas.update(
            {
                _id: doc._id
            },
            {
                $set: {
                    user_id: db.userdetails.findOne({_id:doc.user_id})._id
                }
            }
        );
    }
);
